create schema hobo_adventure; 
use hobo_adventure;

create table hero (

    name char(125),
    Level int not null,
    XP int not null,
    maxHP int not null, 
    damage int not null,
    strength int not null,
    gold int not null,
    primary key (name)

);

create table spells(
    spell_id INT AUTO_INCREMENT PRIMARY KEY,
    name VARCHAR(125) NOT NULL,
    damage INT NOT NULL,
    element CHAR(50),
    gold_price INT NOT NULL,
    hero_name CHAR(125),
    FOREIGN KEY(hero_name) REFERENCES hero(name) ON DELETE CASCADE
);
